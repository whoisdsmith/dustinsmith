-   `"Dustin Smith" "DOB: 03-20-1985"`

-   `"Dustin Smith" | "dsmith@lifeisasimulation.run" | "iamdustinsmith@gmail.com"`

-   `"Dustin Smith" -site:facebook.com -site:linkedin.com`
